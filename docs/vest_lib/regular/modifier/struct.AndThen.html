open spec fn spec_parse(&self, s: Seq<u8>) -> Option<(int, Self::Type)>

{
    if let Some((n, v1)) = self.0.spec_parse(s) {
        if let Some((m, v2)) = self.1.spec_parse(v1) {
            if m == n { Some((n, v2)) } else { None }
        } else {
            None
        }
    } else {
        None
    }
}
