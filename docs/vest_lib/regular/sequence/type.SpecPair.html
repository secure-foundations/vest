open spec fn spec_parse(&self, s: Seq<u8>) -> Option<(int, Self::Type)>

{
    if let Some((n, v1)) = self.fst.spec_parse(s) {
        let snd = (self.snd)(v1);
        if let Some((m, v2)) = snd.spec_parse(s.skip(n as int)) {
            Some((n + m, (v1, v2)))
        } else {
            None
        }
    } else {
        None
    }
}
